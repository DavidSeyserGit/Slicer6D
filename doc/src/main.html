308    def load_model(self):
309        fn, _ = QFileDialog.getOpenFileName(self, "Open STL", "", "STL Files (*.stl *.STL)")
310        if not fn: return
311        self.status.showMessage(f"Loading {os.path.basename(fn)}...")
312        QApplication.processEvents()
313        try:
314            mesh = pv.read(fn)
315            if isinstance(mesh, pv.PolyData):
316                 if mesh.n_cells == 0: raise ValueError("Mesh has no faces.")
317                 faces_conn = mesh.faces
318                 if faces_conn.size > 0:
319                     expected_size = faces_conn[0] + 1
320                     if faces_conn.size % expected_size != 0 or not np.all(faces_conn[::expected_size] == 3):
321                          print("Non-triangular faces detected, attempting triangulation.")
322                          self.base_mesh = mesh.triangulate()
323                     else: self.base_mesh = mesh
324                 else: self.base_mesh = mesh.triangulate()
325            elif isinstance(mesh, (pv.UnstructuredGrid, pv.MultiBlock)):
326                 print("Loaded complex type, extracting/merging surface geometry.")
327                 self.base_mesh = mesh.extract_geometry().triangulate()
328            else: raise TypeError(f"Unsupported mesh type: {type(mesh)}")
329            if not isinstance(self.base_mesh, pv.PolyData) or self.base_mesh.n_cells == 0:
330                 raise ValueError("Failed to obtain a valid surface mesh.")
331        except Exception as e:
332            self.status.showMessage(f"Error loading/processing STL: {e}")
333            traceback.print_exc()
334            self.base_mesh = None; self.original_mesh = None; self.deformed_mesh = None
335            self.overhang_faces = None; self.print_bed = None
336            self._reset_slice_data()
337            self.update_display(); return
338        self.original_mesh = self.base_mesh.copy()
339        self.deformed_mesh  = None; self.overhang_faces = None
340        self._reset_slice_data()
341        self.rot_x_sb.blockSignals(True); self.rot_x_sb.setValue(0); self.rot_x_sb.blockSignals(False)
342        self.rot_y_sb.blockSignals(True); self.rot_y_sb.setValue(0); self.rot_y_sb.blockSignals(False)
343        self.rot_z_sb.blockSignals(True); self.rot_z_sb.setValue(0); self.rot_z_sb.blockSignals(False)
344        try: self.print_bed = deformer.determine_print_bed(self.original_mesh)
345        except Exception as e:
346            self.status.showMessage(f"Warning: Could not determine print bed: {e}")
347            center = self.original_mesh.center; z_min = self.original_mesh.bounds[4]
348            self.print_bed = ([0, 0, 1], [center[0], center[1], z_min])
349        self.mesh_vis_cb.setChecked(True)
350        self.update_display()
351        self.plotter.view_isometric(); self.plotter.reset_camera(render=True)
352        self.status.showMessage(f"Loaded: {os.path.basename(fn)} ({self.original_mesh.n_cells} faces)")

389    def detect_overhangs(self):
390        if self.original_mesh is None: self.status.showMessage("Load a model first."); return
391        if self.print_bed is None: self.status.showMessage("Print bed not determined."); return
392        mesh_to_check = self.original_mesh; current_print_bed = self.print_bed
393        angle = self.ang_sb.value(); support_dist = self.sup_sb.value(); use_ray = self.ray_cb.isChecked()
394        self.status.showMessage("Detecting overhangs..."); QApplication.processEvents()
395        try:
396            faces = deformer.detect_overhangs(mesh_to_check, angle_threshold=angle, print_bed=current_print_bed)
397            if use_ray and faces.size > 0:
398                self.status.showMessage(f"Detected {faces.size} candidates, filtering by ray trace..."); QApplication.processEvents()
399                centers = mesh_to_check.cell_centers().points; filtered_faces = []
400                ray_direction = -np.array(current_print_bed[0])
401                for f_idx in faces:
402                    if f_idx >= mesh_to_check.n_cells: continue
403                    center_point = centers[f_idx]; start_point = center_point + ray_direction * 1e-4
404                    hit_points, hit_indices = mesh_to_check.ray_trace(start_point, ray_direction, first_point=False)
405                    is_supported = False
406                    if len(hit_points) > 0:
407                        distances = np.linalg.norm(hit_points - center_point, axis=1); min_dist = np.min(distances)
408                        if min_dist <= support_dist: is_supported = True
409                    if not is_supported: filtered_faces.append(f_idx)
410                faces = np.array(filtered_faces, dtype=int)
411            self.overhang_faces = faces; self.update_display()
412            if faces.size > 0: self.status.showMessage(f"Detected {faces.size} unsupported overhang faces.")
413            else: self.status.showMessage("No unsupported overhangs detected.")
414        except Exception as e:
415            self.status.showMessage(f"Overhang detection error: {e}"); traceback.print_exc()
416            self.overhang_faces = None; self.update_display()

418    def deform_model(self):
419        if self.original_mesh is None: self.status.showMessage("Load a model first."); return
420        if self.print_bed is None: self.status.showMessage("Print bed not determined."); return
421        mesh_to_deform = self.original_mesh; current_print_bed = self.print_bed
422        angle = self.ang_sb.value(); support = self.sup_sb.value(); use_ray = self.ray_cb.isChecked(); max_iter= 50

470        tmp_stl_file = None
471        try:
472            self.status.showMessage(f"Preparing {source_name} mesh for slicing..."); QApplication.processEvents()
473            with tempfile.NamedTemporaryFile(suffix=".stl", delete=False) as tmp:
474                tmp_stl_file = tmp.name; mesh_to_slice.save(tmp_stl_file)

497            QApplication.processEvents()

499            if self.slice_optimize_cb.isChecked():

508                except Exception as opt_e:

510            else: self.status.showMessage(f"Raw G-code saved. Optimization skipped.")
511
512            if self.max_layer_index >= 0:
513                self.layer_slider.setEnabled(True)
514                self.layer_slider.setMaximum(self.max_layer_index)
515                # Set slider value without triggering update yet
516                self.layer_slider.blockSignals(True)
517                self.layer_slider.setValue(self.max_layer_index)
518                self.layer_slider.blockSignals(False)
519                # Now trigger the update to show the top layer
520                self._update_layer_view()
521                self.status.showMessage(f"Slice complete. Use slider for layer view or click 'Show Full Toolpaths'.")
522            else: self.status.showMessage("Slicing done, but no layers to view.")
523
524        except Exception as e:
525            self.status.showMessage(f"Slicing/G-code Error: {e}"); traceback.print_exc()
526            self._reset_slice_data()
527        finally:
528            if tmp_stl_file and os.path.exists(tmp_stl_file):
529                try: os.remove(tmp_stl_file)
530                except OSError as ose: print(f"Warning: Could not delete temp file {tmp_stl_file}: {ose}")

576    def _update_layer_view(self):
577        """Updates paths and mesh clipping based on the layer slider."""
578        if not self.layer_slider.isEnabled() or self.generated_slice_paths is None or self.slice_z_levels is None:
579            # If called when not ready (e.g., during reset), just ensure view is clear
580            self._clear_layer_view()
581            self.plotter.render()
582            return
583
584        current_layer_index = self.layer_slider.value()
585        if not (0 <= current_layer_index <= self.max_layer_index): return
586
587        target_z = self.slice_z_levels[current_layer_index]
588        self.is_layer_view_active = True
589        self.layer_slider_label.setText(f"Layer: {current_layer_index + 1} / {self.max_layer_index + 1} (Z={target_z:.3f})")
590
591        # --- Filter Paths ---
592        filtered_paths_segments = []
593        epsilon = 1e-5
594        for path_segment in self.generated_slice_paths:
595             if isinstance(path_segment, np.ndarray) and path_segment.ndim == 2 and path_segment.shape[1] == 3:
596                 if np.max(path_segment[:, 2]) <= target_z + epsilon:
597                     filtered_paths_segments.append(path_segment)
598
599        # --- Clip Mesh ---
600        mesh_to_clip = self.deformed_mesh if self.deformed_mesh is not None else self.original_mesh
601        clipped_mesh = None
602        if mesh_to_clip:
603            try:
604                clipped_mesh = mesh_to_clip.clip(normal=[0, 0, 1], origin=[0, 0, target_z], invert=False, inplace=False)
605            except Exception as clip_e: print(f"Warning: Mesh clipping failed: {clip_e}")
606
607        # --- Update Plotter ---
608        # Remove actors specific to other views FIRST
609        self.plotter.remove_actor(ACTOR_MAIN_MESH, render=False)
610        self.plotter.remove_actor(ACTOR_ORIG_WIREFRAME, render=False) # Explicitly remove wireframe
611        self.plotter.remove_actor(ACTOR_FULL_PATHS, render=False)
612        self.plotter.remove_actor(ACTOR_LAYER_VIEW_PATHS, render=False) # Remove old layer paths
613        self.plotter.remove_actor(ACTOR_LAYER_VIEW_MESH, render=False)  # Remove old layer mesh
614
615        # Add Clipped Mesh Actor
616        if clipped_mesh and clipped_mesh.n_points > 0:
617             style = 'wireframe' if self.wf_cb.isChecked() else 'surface'
618             # Use the color appropriate for the clipped mesh (original or deformed)
619             color = 'cyan' if self.deformed_mesh is not None else 'lightgrey'
620             self.plotter.add_mesh(clipped_mesh, style=style, color=color,
621                                   show_edges=self.ed_cb.isChecked(), edge_color='gray',
622                                   name=ACTOR_LAYER_VIEW_MESH)
623
624        # Add Filtered Paths Actor
625        if filtered_paths_segments:
626            all_points_list = []; all_lines_list = []; point_count = 0
627            for path_segment in filtered_paths_segments:
628                num_points = path_segment.shape[0]; all_points_list.append(path_segment)
629                line_indices = np.arange(point_count, point_count + num_points)
630                lines_this_segment = np.vstack([np.full(num_points - 1, 2), line_indices[:-1], line_indices[1:]]).T.flatten()
631                all_lines_list.append(lines_this_segment); point_count += num_points
632            if all_points_list:
633                try:
634                    combined_points = np.vstack(all_points_list); combined_lines = np.concatenate(all_lines_list)
635                    if combined_points.size > 0 and combined_lines.size > 0:
636                        line_polydata = pv.PolyData(combined_points); line_polydata.lines = combined_lines
637                        z_scalars = combined_points[:, 2]; line_polydata['Z-Height'] = z_scalars
638                        self.plotter.add_mesh(line_polydata, scalars='Z-Height', cmap='viridis',
639                                              line_width=2, name=ACTOR_LAYER_VIEW_PATHS)
640                except Exception as path_build_e: print(f"Error building layer view paths: {path_build_e}")
641
642        self.plotter.render()

644    def update_display(self):

646        actors_to_remove = [

669                                       position="upper_left", name=ACTOR_STATUS_TEXT, color='white')
670
671            # Add Overhangs (if not in layer view)

309    def load_model(self):
310        fn, _ = QFileDialog.getOpenFileName(self, "Open STL", "", "STL Files (*.stl *.STL)")
311        if not fn: return
312        self.status.showMessage(f"Loading {os.path.basename(fn)}...")
313        QApplication.processEvents()
314        try:
315            mesh = pv.read(fn)
316            if isinstance(mesh, pv.PolyData):
317                 if mesh.n_cells == 0: raise ValueError("Mesh has no faces.")
318                 faces_conn = mesh.faces
319                 if faces_conn.size > 0:
320                     expected_size = faces_conn[0] + 1
321                     if faces_conn.size % expected_size != 0 or not np.all(faces_conn[::expected_size] == 3):
322                          print("Non-triangular faces detected, attempting triangulation.")
323                          self.base_mesh = mesh.triangulate()
324                     else: self.base_mesh = mesh
325                 else: self.base_mesh = mesh.triangulate()
326            elif isinstance(mesh, (pv.UnstructuredGrid, pv.MultiBlock)):
327                 print("Loaded complex type, extracting/merging surface geometry.")
328                 self.base_mesh = mesh.extract_geometry().triangulate()
329            else: raise TypeError(f"Unsupported mesh type: {type(mesh)}")
330            if not isinstance(self.base_mesh, pv.PolyData) or self.base_mesh.n_cells == 0:
331                 raise ValueError("Failed to obtain a valid surface mesh.")
332        except Exception as e:
333            self.status.showMessage(f"Error loading/processing STL: {e}")
334            traceback.print_exc()
335            self.base_mesh = None; self.original_mesh = None; self.deformed_mesh = None
336            self.overhang_faces = None; self.print_bed = None
337            self._reset_slice_data()
338            self.update_display(); return
339        self.original_mesh = self.base_mesh.copy()
340        self.deformed_mesh  = None; self.overhang_faces = None
341        self._reset_slice_data()
342        self.rot_x_sb.blockSignals(True); self.rot_x_sb.setValue(0); self.rot_x_sb.blockSignals(False)
343        self.rot_y_sb.blockSignals(True); self.rot_y_sb.setValue(0); self.rot_y_sb.blockSignals(False)
344        self.rot_z_sb.blockSignals(True); self.rot_z_sb.setValue(0); self.rot_z_sb.blockSignals(False)
345        try: self.print_bed = deformer.determine_print_bed(self.original_mesh)
346        except Exception as e:
347            self.status.showMessage(f"Warning: Could not determine print bed: {e}")
348            center = self.original_mesh.center; z_min = self.original_mesh.bounds[4]
349            self.print_bed = ([0, 0, 1], [center[0], center[1], z_min])
350        self.mesh_vis_cb.setChecked(True)
351        self.update_display()
352        self.plotter.view_isometric(); self.plotter.reset_camera(render=True)
353        self.status.showMessage(f"Loaded: {os.path.basename(fn)} ({self.original_mesh.n_cells} faces)")

390    def detect_overhangs(self):
391        if self.original_mesh is None: self.status.showMessage("Load a model first."); return
392        if self.print_bed is None: self.status.showMessage("Print bed not determined."); return
393        mesh_to_check = self.original_mesh; current_print_bed = self.print_bed
394        angle = self.ang_sb.value(); support_dist = self.sup_sb.value(); use_ray = self.ray_cb.isChecked()
395        self.status.showMessage("Detecting overhangs..."); QApplication.processEvents()
396        try:
397            faces = deformer.detect_overhangs(mesh_to_check, angle_threshold=angle, print_bed=current_print_bed)
398            if use_ray and faces.size > 0:
399                self.status.showMessage(f"Detected {faces.size} candidates, filtering by ray trace..."); QApplication.processEvents()
400                centers = mesh_to_check.cell_centers().points; filtered_faces = []
401                ray_direction = -np.array(current_print_bed[0])
402                for f_idx in faces:
403                    if f_idx >= mesh_to_check.n_cells: continue
404                    center_point = centers[f_idx]; start_point = center_point + ray_direction * 1e-4
405                    hit_points, hit_indices = mesh_to_check.ray_trace(start_point, ray_direction, first_point=False)
406                    is_supported = False
407                    if len(hit_points) > 0:
408                        distances = np.linalg.norm(hit_points - center_point, axis=1); min_dist = np.min(distances)
409                        if min_dist <= support_dist: is_supported = True
410                    if not is_supported: filtered_faces.append(f_idx)
411                faces = np.array(filtered_faces, dtype=int)
412            self.overhang_faces = faces; self.update_display()
413            if faces.size > 0: self.status.showMessage(f"Detected {faces.size} unsupported overhang faces.")
414            else: self.status.showMessage("No unsupported overhangs detected.")
415        except Exception as e:
416            self.status.showMessage(f"Overhang detection error: {e}"); traceback.print_exc()
417            self.overhang_faces = None; self.update_display()

452    def slice_model(self):
453        """Slices the model, generates G-code, enables layer slider."""
454        self._reset_slice_data() # Clear previous results first
455
456        mesh_to_slice = None; source_name = ""
457        if self.slice_deformed_cb.isChecked() and self.deformed_mesh is not None:
458            mesh_to_slice = self.deformed_mesh; source_name = "deformed"
459        elif self.original_mesh is not None:
460            mesh_to_slice = self.original_mesh; source_name = "original"
461        else: self.status.showMessage("No mesh loaded or selected to slice."); return
462
463        fn, _ = QFileDialog.getSaveFileName(self, "Save Gcode Base Name", "", "Gcode Files (*.gcode)")
464        if not fn: return
465
466        if not fn.lower().endswith('.gcode'): fn += '.gcode'
467        base_path = os.path.splitext(fn)[0]
468        raw_gcode_path = base_path + "_raw.gcode"
469        opt_gcode_path = base_path + "_opt.gcode"
470
471        tmp_stl_file = None
472        try:
473            self.status.showMessage(f"Preparing {source_name} mesh for slicing..."); QApplication.processEvents()
474            with tempfile.NamedTemporaryFile(suffix=".stl", delete=False) as tmp:
475                tmp_stl_file = tmp.name; mesh_to_slice.save(tmp_stl_file)
476
477            self.status.showMessage(f"Slicing {source_name} mesh..."); QApplication.processEvents()
478            layer_h = self.slice_layer_sb.value(); ext_mm = self.slice_extrude_sb.value()
479            feed = self.slice_feed_sb.value(); travel = self.slice_travel_sb.value()
480            infill_space = self.slice_infill_spacing_sb.value(); infill_angle = self.slice_infill_angle_sb.value()
481
482            z_levels, layer_polygons = slicing.slice_mesh_to_polygons(tmp_stl_file, layer_h)
483            if not z_levels or not layer_polygons or all(not lp for lp in layer_polygons):
484                 self.status.showMessage("Slicing completed, but no valid layers found.")
485                 self._reset_slice_data(); return
486
487            self.slice_z_levels = z_levels; self.slice_layer_polygons = layer_polygons
488            self.max_layer_index = len(z_levels) - 1
489
490            self.status.showMessage("Generating G-code with infill..."); QApplication.processEvents()
491            raw_gcode_string, generated_paths = slicing.generate_gcode_with_infill(
492                z_levels, layer_polygons, feedrate=feed, extrusion_per_mm=ext_mm,
493                travel_feed=travel, infill_spacing=infill_space, infill_angle=infill_angle)
494            self.generated_slice_paths = generated_paths
495
496            with open(raw_gcode_path, "w") as f: f.write(raw_gcode_string)
497            self.status.showMessage(f"Wrote raw G-code to {os.path.basename(raw_gcode_path)}")
498            QApplication.processEvents()
499
500            if self.slice_optimize_cb.isChecked():
501                self.status.showMessage("Optimizing G-code..."); QApplication.processEvents()
502                try:
503                    raw_lines = raw_gcode_string.splitlines()
504                    if hasattr(gcode_optimizer, 'optimize_gcode'):
505                        opt_lines = gcode_optimizer.optimize_gcode(raw_lines)
506                        with open(opt_gcode_path, "w") as f: f.write("\n".join(opt_lines))
507                        self.status.showMessage(f"Wrote optimized G-code to {os.path.basename(opt_gcode_path)}.")
508                    else: self.status.showMessage("G-code optimizer function not found. Skipping.")
509                except Exception as opt_e:
510                    self.status.showMessage(f"G-code optimization failed: {opt_e}"); traceback.print_exc()
511            else: self.status.showMessage(f"Raw G-code saved. Optimization skipped.")
512
513            if self.max_layer_index >= 0:
514                self.layer_slider.setEnabled(True)
515                self.layer_slider.setMaximum(self.max_layer_index)
516                # Set slider value without triggering update yet
517                self.layer_slider.blockSignals(True)
518                self.layer_slider.setValue(self.max_layer_index)
519                self.layer_slider.blockSignals(False)
520                # Now trigger the update to show the top layer
521                self._update_layer_view()
522                self.status.showMessage(f"Slice complete. Use slider for layer view or click 'Show Full Toolpaths'.")
523            else: self.status.showMessage("Slicing done, but no layers to view.")
524
525        except Exception as e:
526            self.status.showMessage(f"Slicing/G-code Error: {e}"); traceback.print_exc()
527            self._reset_slice_data()
528        finally:
529            if tmp_stl_file and os.path.exists(tmp_stl_file):
530                try: os.remove(tmp_stl_file)
531                except OSError as ose: print(f"Warning: Could not delete temp file {tmp_stl_file}: {ose}")
